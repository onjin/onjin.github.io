home.pl obsługuje pythona jako cgi (pliki .py) więc tego będziemy się trzymać. Instalację wykonamy w katalogu /py

instalacja django

hardcoded ale działa ;) wrzucamy to do /py i ruchamiamy poprzez www (plik install.py)

 1  #!/usr/bin/env python
 2 
 3  import os
 4 
 5  os.system("wget http://www.djangoproject.com/download/1.0.2/tarball/")

 7  os.system("rm Django-1.0.2-final.tar.gz")
 8  os.system("mv Django-1.0.2-final tmp")
 9  os.system("mv tmp/django django")

nasz projekt

$ django-admin.py startproject pytest

dispatch.py

 1  #!/usr/bin/env python
 2 
 3  import os, sys
 4  import django.core.handlers.wsgi
 5 
 6  def run_with_cgi(application):
 7 
 8      environ                      = dict(os.environ.items())
 9      environ['wsgi.input']        = sys.stdin
10      environ['wsgi.errors']       = sys.stderr
11      environ['wsgi.version']      = (1,0)
12      environ['wsgi.multithread']  = False
13      environ['wsgi.multiprocess'] = True
14      environ['wsgi.run_once']     = True
15 
16      if environ.get('HTTPS','off') in ('on','1'):
17          environ['wsgi.url_scheme'] = 'https'
18      else:
19          environ['wsgi.url_scheme'] = 'http'
20 
21      headers_set  = []
22      headers_sent = []
23 
24      def write(data):
25          if not headers_set:
26              raise AssertionError("write() before start_response()")
27 
28          elif not headers_sent:
29              # Before the first output, send the stored headers
30              status, response_headers = headers_sent[:] = headers_set
31              sys.stdout.write('Status: %s\\r\\n' % status)
32              for header in response_headers:
33                  sys.stdout.write('%s: %s\\r\\n' % header)
34              sys.stdout.write('\\r\\n')
35 
36          sys.stdout.write(data)
37          sys.stdout.flush()
38 
39      def start_response(status,response_headers,exc_info=None):
40          if exc_info:
41              try:
42                  if headers_sent:
43                      # Re-raise original exception if headers sent
44                      raise exc_info[0], exc_info[1], exc_info[2]
45              finally:
46                  exc_info = None     # avoid dangling circular ref
47          elif headers_set:
48              raise AssertionError("Headers already set!")
49 
50          headers_set[:] = [status,response_headers]
51          return write
52 
53      result = application(environ, start_response)
54      try:
55          for data in result:
56              if data:    # don't send headers until body appears
57                  write(data)
58          if not headers_sent:
59              write('')   # send headers now if body was empty
60      finally:
61          if hasattr(result,'close'):
62          result.close()
63 
64 
65  # Change this to the directory above your site code.
66  sys.path.append("/py")
67 
68  # Change mysite to the name of your site package
69  os.environ['DJANGO_SETTINGS_MODULE'] = 'pytest.settings'
70 
71  run_with_cgi(django.core.handlers.wsgi.WSGIHandler())
